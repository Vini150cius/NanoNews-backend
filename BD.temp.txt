create table public.articles (
  id bigint generated by default as identity not null,
  uuid_author uuid not null default auth.uid (),
  headline text not null,
  header_image text null,
  tags text null,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone null,
  constraint articles_pkey primary key (id)
) TABLESPACE pg_default;


create table public.content_of_article (
  id bigint generated by default as identity not null,
  id_article bigint not null,
  "order" bigint not null,
  type_component text not null,
  content text not null,
  constraint content_of_article_pkey primary key (id),
  constraint content_of_article_id_article_fkey foreign KEY (id_article) references articles (id) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;
